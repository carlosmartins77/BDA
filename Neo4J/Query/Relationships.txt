-- Publicacao
MATCH
  (a:Utilizador),
  (b:Publicacoes)
WHERE a.IdUtilizador = b.IdUtilizadorFK
CREATE (a)-[p:Published]->(b)

-- Logs
MATCH
  (a:Logs),
  (b:LogType)
WHERE a.IdLogTypeFK = b.IdLogType
CREATE (a)-[l:log_Type]->(b)

-- Equipa

MATCH
  (a:Utilizador),
  (b:Equipa)
WHERE a.IdUtilizador = b.IdUtilizadorCapitaoFK
CREATE (a)-[t:Team_Leader]->(b)

  (a:JogoFutebol),
  (b:Equipa)

  (a:JogoFutebol),
